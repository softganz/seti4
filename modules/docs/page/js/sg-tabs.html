$(document).on('click', '.sg-tabs>ul.tabs>li>a', function(e) { var $this=$(this) var $parent=$this.closest('.sg-tabs') var href=$this.attr('href') $this.closest('ul').children('li').removeClass('active') $this.closest('li').addClass('active') if ($this.attr('target')!=undefined) return true; if (href.left(1)=='#') { $parent.children('div').hide() $parent.children($this.attr('href')).show() } else { notify('กำลังโหลด') //window.history.pushState({},$this.text(),href) $.get(href,function(html) { $parent.children('div').html(html) notify() }) } return false });
