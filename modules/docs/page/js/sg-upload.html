$(document).on('change', "form.sg-upload .inline-upload", function() { var $this=$(this) var $form=$this.closest("form") var target=$form.data('rel') console.log('Inline upload file start and show result in '+target) notify(" กำลังอัพโหลดไฟล์ กรุณารอสักครู่....") $form.ajaxForm({ success: function(data) { if (target) { if ($form.data('append')) { var insertElement='<'+$form.data('append')+'>'+data+'' $('#'+target).append(insertElement) } else if ($form.data('prepend')) { var insertElement='<'+$form.data('prepend')+'>'+data+'' $('#'+target).prepend(insertElement) } else { $('#'+target).html(data) } } notify("ดำเนินการเสร็จแล้ว.",5000) $this.val("") $this.replaceWith($this.clone(true)) } }).submit() });
